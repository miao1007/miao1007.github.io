int dfs(TreeNode t){
    if(t == null){
        return 0;
    }
    if(t.left == null && t.right == null){
        return 1;
    }
    int l = dfs(t.left);
    int r = dfs(t.right);
    if(l == -1 || r == -1 ||Math.abs(l - r) > 1 ){
        return -1;
    }
    return Math.max(l,r) + 1;
}

public int findTilt(TreeNode root) {
    if(root == null){
        return 0;
    }
    int a = sum(root.left);
    int b = sum(root.right);
    return Math.abs(a - b) + findTilt(root.left) + findTilt(root.right);
}

int sum(TreeNode t){
    if(t == null){
        return 0;
    }
    int l = sum(t.left);
    int r = sum(t.right);
    return t.val + l + r;
}

class Solution {
    public boolean isSubtree(TreeNode s, TreeNode t) {
        if(s == null){
            return s == t;
        }
        return dfs(s, t) || isSubtree(s.left, t) || isSubtree(s.right, t);
    }
    
    boolean dfs(TreeNode s, TreeNode t){
        if(s == null && t == null){
            return true;
        }
        // one of null
        if(s == null || t == null){ 
            return false;
        }
        if(s.val == t.val){
            return dfs(s.left, t.left) && dfs(s.right, t.right);
        }
        return false;
    }
}

void hash(TreeNode s,StringBuilder sb){

    sb.append("{},");
    return;

void dfs(Node n, List<Integer> l){

  }

// post Order，适用于对比/bottom up求和等场景
void dft(Node n){
  if(n == null){
    return;
  }
  for(Node c: n.children){
    dft(c);
  }
  l.add(n.val);
}
